# fortnum test suite. Each test is one self-contained block: build an
# executable, link fortnum, register with add_test. Downstream agents append
# one block per new test below.

add_executable(test_fortnum_smoke test_fortnum_smoke.f90)
target_link_libraries(test_fortnum_smoke PRIVATE fortnum)
add_test(NAME fortnum_smoke COMMAND test_fortnum_smoke)

add_executable(test_fortnum_kinds test_fortnum_kinds.f90)
target_link_libraries(test_fortnum_kinds PRIVATE fortnum)
add_test(NAME fortnum_kinds COMMAND test_fortnum_kinds)

add_executable(test_fortnum_status test_fortnum_status.f90)
target_link_libraries(test_fortnum_status PRIVATE fortnum)
add_test(NAME fortnum_status COMMAND test_fortnum_status)

# Oracle harness self-test. The seeded reference table lives in the source
# tree (regenerated by test/oracle/gen_oracle.py); pass its absolute path so
# the test is independent of the ctest working directory.
add_executable(test_oracle_selftest oracle/test_oracle_selftest.f90)
target_link_libraries(test_oracle_selftest PRIVATE fortnum)
add_test(NAME oracle_selftest
    COMMAND test_oracle_selftest
        ${CMAKE_CURRENT_SOURCE_DIR}/oracle/data/gamma.csv)
